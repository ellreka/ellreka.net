## コード

```ts
interface Options {
  foo?: string
  bar?: string
}

const func = ({ foo = 'foo', bar = 'bar' }: Options = {}) => {
  return foo + bar
}

func()
// => foobar

func({})
// => foobar

func({ foo: 'bar' })
// => barbar

func({ foo: 'foo', bar: 'foo' })
// => foofoo
```
